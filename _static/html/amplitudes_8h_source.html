50  throw mspass::utility::MsPASSError("scale function: illegal perf level specified for clip percentage scale - must be between 0 and 1\nData unaltered - may cause downstream problems",
51  mspass::utility::ErrorSeverity::Suspect);
52  try{
53  double newcalib(1.0);
54  if(d.is_defined(scale_factor_key))
55  {
56  newcalib=d.get_double(scale_factor_key);
57  }
58  double amplitude,dscale;
59  switch(method)
60  {
61  case ScalingMethod::Peak:

72  break;
73  case ScalingMethod::MAD:

77  break;
78  case ScalingMethod::RMS:

81  dscale = level/amplitude;
82  newcalib /= dscale;
83  };

158  {
159  double amplitude;

162  {

168  break;
169  case ScalingMethod::MAD:
170  amplitude=MADAmplitude(*dptr);
171  break;
172  case ScalingMethod::RMS:
173  default:
174  amplitude=RMSAmplitude(*dptr);
175  };
176  ++nlive;
177  amps.push_back(log(amplitude));
178  }
179  /*Silently return a 0 if there are no live data members*/
180  if(nlive==0) return 0.0;

182  if(use_mean)

185  }
186  else

189  }
190

192  avgamp=exp(avgamp);
193  double dscale=level/avgamp;
194  /* Now scale the data and apply calib */
195  for(dptr=d.member.begin();dptr!=d.member.end();++dptr)
196  {
197  if(dptr->live())
198  {
199  double calib;
200  (*dptr) *= dscale;

204  }

207  calib=1.0;
208  }
209  calib/=dscale;

211  }
212  }
213  return avgamp;
